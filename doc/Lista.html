def initialize(nodes)
  for i in 0..nodes.length-1 do
    nodes[i]=Node.new(nodes[i],nil,nil)
  end
  
  j = nodes.length
  j -= 1
  @cola = nodes[0]
  @cabeza = nodes[j]
  if j > 1
    @cola.next = nodes[1]
    for i in 1..j-1 do
      nodes[i].next = nodes[i+1]
      nodes[i].prev = nodes[i-1]
    end
    @cabeza.prev = nodes[j-1]
  else 
    if j == 1
      @cabeza.prev = @cola
      @cola.next = @cabeza
    else
      @cabeza = @cola
    end
  end
 
end
